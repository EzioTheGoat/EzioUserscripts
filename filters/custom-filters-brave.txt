! =============================================================================
! SECTION III: SITE-SPECIFIC AD & CONTENT BLOCKING RULES
! -----------------------------------------------------------------------------
! Description:
!   The rules in this section target unwanted elements on ArabSeed-related
!   sites such as "clickme" buttons, ad overlays, and promotional download
!   buttons. Many rules use the "upward" operator (:upward()) to remove the
!   parent element that contains unwanted ads or overlays. In addition, for
!   each site the dynamic download‑button blocker is applied if the URL is “bare.”
! -----------------------------------------------------------------------------

! ============================================================
! m.gamehub.cam
! ------------------------------------------------------------
m.gamehub.cam###clickme > .btn-collab.hover-top.btn
!m.gamehub.cam##.first > .DownloadMainContent > .InnerDownload > .Adabove:not(:has(a[href*="&t=1"])):upward(.Adabove)
m.gamehub.cam##.visible-block
m.gamehub.cam##.relativeee.DownloadArea > .DownloadBlock > .TitleCenteral [href^="https://fulvideozrt.click/"] > .desktop-button
m.gamehub.cam##.relativeee.DownloadArea > .DownloadBlock > .DownloadBlock > .TitleCenteral [href^="https://fulvideozrt.click/"] > .desktop-button
m.gamehub.cam##.relativeee.DownloadArea > .DownloadBlock > .DownloadBlock > .DownloadBlock > .TitleCenteral [href^="https://fulvideozrt.click/"] > .desktop-button
m.gamehub.cam##.DownloadBlock > .DownloadBlock > .DownloadBlock > .DownloadBlock > .TitleCenteral [href^="https://fulvideozrt.click/"] > .desktop-button
m.gamehub.cam##.desktop-button
m.gamehub.cam##a#btn[href*="?r="]:not([href*="&t=1&gfs=1&tfs=1"])
m.gamehub.cam##a#btn[href^="https://m.gamehub.cam/"][href*='?r='] > .btn-collab.hover-top.btn
